#VB
   dim modcliente
   dim rc,dbo
   dim f,i
   set db0=Ambiente.GetObject("DB")
   set rc=db0.openview("select a.cod as codigo,a.des as descr,a.um as um,a.costo as custo,cm.des as descategoria from articoli a left join tabelle cm on a.catmer=cm.cod and cm.tip='cm' where a.flmag=1")
   f=""
   If rc.EOF = False Then
msgbox "asrfasdf"
      for i=0 to rc.Count-1
         f=f & rc.cnome(i) & ";"
      Next
      f=f & vbcrlf
      Do Until rc.EOF
         Ambiente.xmessaggio "Exportando " & rc.s(0)
         for i=0 to rc.Count-1
            f = f & replace(rc.s(i),",",".") & ";"
         Next
         f=f & vbcrlf
         rc.MoveNext
      Loop
   End If
   
   set rc=nothing
   dim fso
   set fso=createobject("Scripting.filesystemobject")
   if fso.folderexists(v("pasta_salva"))=false then
      fso.CreateFolder v("pasta_salva")
   end if
   scrivifiletesto v("pasta_salva") & "\itens.csv",f
   
   set db0=nothing
   set fso=nothing
   If err <> 0 Then
      MsgBox err.description
   Else
      msgbox "Criado arquivo " & v("pasta_salva") & "\itens.csv"
   End If
#ENDVB
