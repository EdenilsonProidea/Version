// 0 - codigo
// 1 - descriçăo
// 2 - unidade de medida
// 3 - custo
// 4 - família de produto
// 5 - descricao da familia de produto
// 6 - largura
// 7 - altura
// 8 - profundidade
// 9 - flag Ativo Sim/Năo (ou 0 e 1) isso seria interessante ter se possível
// 10 - Flag Excluir (0 e 1)
// 11 - PESO liquido
// 12 - PESO bruto

#VB
   dim rc,dbo
   dim f,i
   set db0=Ambiente.GetObject("DB")
   set rc=db0.openview("select a.cod as codigo,a.des as descr,a.um as um,a.costo as custo,a.catmer as categoria,cm.des as descategoria,a.diml as largura,a.dima as altura, a.dimp as profundidade,1 as ativo,0 as FlagExcluir,a.peson as peso from articoli a left join tabelle cm on a.catmer=cm.cod and cm.tip='cm' where a.flmag=1 " )
   f=""
   If rc.EOF = False Then
      for i=0 to rc.Count-1
         f=f & """" & rc.cnome(i) & """;"
      Next
      f=f & vbcrlf
      Do Until rc.EOF
         Ambiente.xmessaggio "Exportando " & rc.s(0)
         for i=0 to rc.Count-1
            f = f & replace(rc.s(i),",",".") & ";"
         Next
         f=f & vbcrlf
         rc.MoveNext
      Loop
   End If
   
   set rc=nothing
   dim fso
   set fso=createobject("Scripting.filesystemobject")
   if fso.folderexists(v("pasta_salva"))=false then
      fso.CreateFolder v("pasta_salva")
   end if
   scrivifiletesto v("pasta_salva") & "\itens.csv",f
   set rc=db0.openview("select a.cod as cod,a.des as des,a.um as um,a.catmer as categoria,cm.des as descategoria from articoli a left join tabelle cm on a.catmer=cm.cod and cm.tip='cm' where (a.modello='' or a.modello is null) or flmag=1  " )
   f=""
   If rc.EOF = False Then
      
      Do Until rc.EOF
         Ambiente.xmessaggio "Exportando " & rc.s(0)
         for i=0 to rc.Count-1
            f = f  & replace(rc.s(i),",",".") & ";"
         Next
         f=f & vbcrlf
         rc.MoveNext
      Loop
   End If
   
   set rc=nothing
   
   scrivifiletesto v("pasta_salva_html"),f
   set db0=nothing
   set fso=nothing
   If err <> 0 Then
      MsgBox err.description
   Else
      msgbox "Criado arquivo " & v("pasta_salva") & "\itens.csv"
   End If
#ENDVB
